def ols(y, X):
    XtX_inv = np.linalg.inv(X.T @ X)
    beta    = XtX_inv @ X.T @ y
    y_hat   = X @ beta
    resid   = y - y_hat
    return beta, resid, y_hat

def prais_winsten(y, X, max_iter=50, tol=1e-8):
    n, k = X.shape
    _, resid, _ = ols(y, X)
    rho = 0.0
    for _ in range(max_iter):
        rho_new = np.corrcoef(resid[1:], resid[:-1])[0, 1]
        y_star  = np.empty(n); X_star = np.empty((n, k))
        y_star[0] = np.sqrt(1 - rho_new**2) * y[0]
        X_star[0] = np.sqrt(1 - rho_new**2) * X[0]
        y_star[1:] = y[1:] - rho_new * y[:-1]
        X_star[1:] = X[1:] - rho_new * X[:-1]
        beta_new, _, yhat_new = ols(y_star, X_star)
        resid_orig = y - X @ beta_new
        if abs(rho_new - rho) < tol:
            break
        rho = rho_new; resid = resid_orig
    return beta_new, resid_orig, X @ beta_new
